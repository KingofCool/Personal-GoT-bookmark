# AI looks for clan annexation targets
character_event = {
	id = jogos_nhai.1
	
	hide_window = yes
	is_triggered_only = yes
	only_independent = yes
	prisoner = no
	capable_only = yes
	min_age = 16
	culture = jogos_nhai
	
	trigger = {
		is_nomadic = yes
		is_ill = no
		prestige = 250
		
		NOT = { trait = content }
	}
	
	
	immediate = {
		random_independent_ruler = {
			limit = {
				independent = yes	
				is_nomadic = yes
				culture = jogos_nhai
				NOT = { character = ROOT }
				NOT = { has_character_modifier = jhattar_of_jogos_nhai }
				custom_tooltip = {
					text = TOOLTIPjogos_nhai_annex_clanNEIGHBOUR
					hidden_tooltip = {
						any_realm_province = {
							any_neighbor_province = {
								owner = { same_realm = ROOT }
							}
						}
					}
				}
				OR = {
					ai = no
					opinion = { who = ROOT value = 0 }
					AND = {
						prisoner = yes
						host = { character = ROOT }
					}
				}	
				custom_tooltip = {
					text = TOOLTIPjogos_nhai_annex_clanCLAN
					hidden_tooltip = {
						NOT = {
							any_vassal = {
								is_nomadic = yes
								ai = yes
								NOT = { opinion = { who = ROOT value = 0 } }
							}
						}
					}
				}
				OR = {
					prisoner = no
					host = { character = ROOT }
				}				
				NOT = { trait = incapable }
				war = no
				NOT = { reverse_has_opinion_modifier = { who = ROOT modifier = opinion_rejected_jogos_nhai_annex_clan } }
				
				OR = {
					ROOT = { trait = ambitious }
					AND = {
						ROOT = { num_of_count_titles_in_realm = 10 }
						NOT = { num_of_count_titles_in_realm = 8 }
					}	
					is_close_relative = ROOT
					is_allied_with = ROOT
					has_non_aggression_pact_with = ROOT
					is_friend = ROOT
					opinion = { who = ROOT value = 50 }
				}
			}
			character_event = { id = jogos_nhai.2 tooltip = TOOLTIPjogos_nhai.2 days = 5 }
		}
	}
	
	option = {
		name = OK
	}
}
