province_event = {
	id = mountain_clans.14
	desc = "EVTDESCmountain_clans.14"
	picture = GFX_evt_mountain_clans

	trigger = {
		region = world_mountains_of_moon
		OR = {
			NOT = { has_global_flag = war_cooldown_period } 
			had_global_flag = { flag = war_cooldown_period days = 25 } 
		}
		k_mountain_clans = { has_holder = no }
		any_independent_ruler = {
			culture = moon_clansman
			NOT = { demesne_size = 1 }
			ai = yes
			NOT = { has_character_flag = clansmen_hunt_duel_target }
		}
		NOT = {
			has_game_rule = {
				name = non_lore_invasions
				value = off
			}
		}
	}
	
	mean_time_to_happen = {
		years = 4000
		
		modifier = {
			factor = 3
			has_game_rule = {
				name = non_lore_invasions
				value = rare
			}
		}
		modifier = {
			factor = 0.5
			owner = {
				top_liege = {
					primary_title = {
						OR = {
							title = e_vale
							title = k_vale
						}
					}	
				}
			}
		}
		modifier = {
			factor = 0.5
			owner = {
				top_liege = {
					war = yes	
				}
			}
		}
		modifier = {
			factor = 0.5
			e_iron_throne = { holder_scope = { war = yes } }
		}
	}

	option = {
		name = "EVTOPTAmountain_clans.14"
		create_random_soldier = {
			attributes = {
				martial = 9
				diplomacy = 4
				learning = 2
			}
			female = no
			employer = e_rebels
			culture = moon_clansman
			religion = mountain_old_gods
			random_traits = yes
			dynasty = random
			trait = brave
			trait = ambitious
			trait = skilled_tactician
			historical = yes
		}
		custom_tooltip = { text = TOOLTIPmountain_clans.14 }
		new_character = {
			prestige = 600
			random_list = {
				25 = { add_trait = trained_warrior }
				75 = { add_trait = skilled_warrior }
				25 = { add_trait = master_warrior }
			}
			remove_trait = inbred
			remove_trait = imbecile
			remove_trait = infirm
			remove_trait = incapable
			remove_trait = dwarf
			remove_trait = hunchback
			remove_trait = clubfooted
			remove_trait = weak
			remove_trait = leper
			remove_trait = possessed
			remove_trait = ill
			remove_trait = pneumonic
			remove_trait = maimed
			set_character_flag = moon_clansman_invader
			random_independent_ruler = {
				limit = {
					culture = moon_clansman
					NOT = { demesne_size = 1 }
					ai = yes
					NOT = { has_character_flag = clansmen_hunt_duel_target }
				}
				abdicate_to = PREV	
			}
			activate_title = { title = k_mountain_clans status = yes }
			k_mountain_clans = {				
				gain_title = prev
			}
			character_event = { id = mountain_clans.15 days = 2  }			
		}
		hidden_tooltip = {
			any_playable_ruler = {
				limit = { 
					ai = no 
					NOT = {
						any_demesne_province = {
							province_id = ROOT
						}
					}

						capital_scope = { region = world_westeros }
						ROOT = { owner = { same_realm = PREVPREV } }
					}
