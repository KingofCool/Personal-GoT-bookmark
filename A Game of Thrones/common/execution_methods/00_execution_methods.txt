execution_method = {
	death = death_execution_burning
	weight_multiplier = {
		factor = 150
		modifier = {
			factor = 2
			is_female = yes
		}
		modifier = {
			factor = 2
			trait = kinslayer
		}
		modifier = {
			factor = 2
			FROM = {
				ROOT = { just_execution_reason_PREV_trigger = yes }
			}
		}
	}
	trigger = {
		FROM = {
			OR = {
				trait = fire_obsessed

						trait = favored_by_rhllor
						trait = chosen_by_rhllor
					}	
				}
			}
		}
	}
}
