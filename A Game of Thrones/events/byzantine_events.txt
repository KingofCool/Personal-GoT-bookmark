# Jailer Blind
character_event = {
	id = LoR.830
	desc = "EVTDESCLoR.830"
	picture = "GFX_evt_torture"
	
	is_triggered_only = yes
	ai = no
	
	option = {
		name = "EVTOPTALoR.830"
		tooltip = { FROM = { add_trait = blinded } }
		if = {
			limit = {
				FROM = { is_ruler = yes }
			}
			chronicle = {
				entry = CHRONICLE_BLINDED_RULER
				portrait = [From.GetID]
			}
			break = yes
		}
		if = {
			limit = {
				FROM = {
					OR = {
						dynasty = ROOT
						is_close_relative = ROOT
						is_married = ROOT
					}
				}
			}
			chronicle = {
				entry = CHRONICLE_BLINDED_RELATIVE
				portrait = [From.GetID]
			}
			break = yes
		}
		if = {
			limit = {
				FROM = {
					liege = {
						OR = {
							dynasty = PREV
							is_close_relative = PREV
							is_married = PREV
						}
					}	
				}
			}
			chronicle = {
				entry = CHRONICLE_BLINDED_RULER_RELATIVE
				portrait = [From.GetID]
			}
			break = yes
		}
		if = {
			limit = {
				FROM = { trait = dragon_rider }
			}
			chronicle = {
				entry = CHRONICLE_BLINDED_DRAGON_RIDER
				portrait = [From.GetID]
			}
			break = yes
		}
		if = {
			limit = {
				FROM = { trait = kingsguard }
			}
			chronicle = {
				entry = CHRONICLE_BLINDED_KINGSGUARD
				portrait = [From.GetID]
			}
			break = yes
		}
		if = {
			limit = {
				FROM = { 
					OR = {
						prestige = 250
						any_claim = { higher_tier_than = COUNT }
					}
				}
			}
			chronicle = {
				entry = CHRONICLE_BLINDED_CHARACTER
				portrait = [From.GetID]
			}
			break = yes
		}
	}
}

# Prisoner Blind
character_event = {
	id = LoR.831
	desc = "EVTDESCLoR.831"
	picture = "GFX_evt_torture"
	
	show_from_from = yes
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTALoR.831"
		add_trait = blinded
		if = {
			limit = { 
				has_game_rule = {
					name = punishment_release_prisoner
					value = on
				}
			}
			prisoner = no
		}
		hidden_tooltip = {
			if = {
				limit = {
					is_ruler = yes
					has_nickname = no
					is_feudal = yes
					higher_tier_than = DUKE
					independent = yes
					OR = {
						tier = EMPEROR
						primary_title = { NOT = { check_variable = { which = "de_facto_empire" value = 1 } } }
					}
					OR = {
						tier = KING
						culture_group = andal
						culture_group = first_men
						culture_group = free_folk
						culture_group = iron_isles_culture
						primary_title = { title = e_iron_throne }
					}
					NOT = { culture_group = hyrkoon_group }
					NOT = { culture_group = dornish }
				}					
				random = {
					chance = 20
					if = {
						limit = { is_female = no }
						give_nickname = nick_the_blind_king
					}	
					if = {
						limit = { is_female = yes }
						give_nickname = nick_the_blind_queen
					}	
				}
			}
			if = {
				limit = { ai = no }
				chronicle = {
					entry = CHRONICLE_BLINDED
					portrait = [FromFrom.GetID]
				}	
			}
			event_target:chronicle_player = {
				if = {
					limit = { 
						NOT = { character = ROOT }
						OR = {		
							dynasty = ROOT
							is_close_relative = ROOT
							is_liege_or_above = ROOT
							AND = {
								ROOT = { trait = kingsguard }
								has_landed_title = e_iron_throne
							}
						}
					}
					chronicle = {
						entry = CHRONICLE_RELATIVE_BLINDED
						portrait = [Root.GetID]
					}	
				}
			}
		}
	}
}
