			limit =  { is_married = yes }
			remove_spouse = spouse
		}
		if = {
			limit =  { trait = hedonist }
			remove_trait = hedonist
		}	
		if = {
			limit =  { trait = lustful }
			remove_trait = lustful
		}
		if = {
			limit =  { trait = gluttonous }
			remove_trait = gluttonous
		}
		if = {
			limit =  { trait = greedy }
			remove_trait = greedy
		}
		if = {
			limit =  { trait = cynical }
			remove_trait = cynical
		}
		if = {
			limit =  { trait = drunkard }
			remove_trait = drunkard
		}
		if = {
			limit =  { trait = slothful }
			remove_trait = slothful
		}
		if = {
			limit =  { trait = maester }
			remove_trait = maester
			remove_special_character_title = title_maester
		}
		if = {
			limit =  { trait = archmaester }
			remove_trait = archmaester

			limit =  { NOT = { trait = chaste } }

			limit =  { NOT = { trait = diligent } }
