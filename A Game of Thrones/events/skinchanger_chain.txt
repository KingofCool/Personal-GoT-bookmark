#Multi animal check
character_event = {
	id = skinchanger.699
	
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		if = {
			limit = { NOT = { has_character_flag = multi_animal_skinchanger } }
			random = { #1 in 10 skinchangers can have more than one animal
				chance = 10
				set_character_flag = multi_animal_skinchanger
				set_variable = { which = "animal_companions" value = 0.1 }
			}
		}
		if = {
			limit = { has_character_flag = multi_animal_skinchanger }
			change_variable = { which = "animal_companions" value = 1 }
			#50% chance for each extra animal after 2, upto 6
			if = { #6
				limit = { check_variable = { which = "animal_companions" value = 6 } }
				clr_character_flag = multi_animal_skinchanger
			}
			if = { #2-5
				limit = { 
					check_variable = { which = "animal_companions" value = 2 } 
					NOT = { check_variable = { which = "animal_companions" value = 6 } }
				}
				random = {
					chance = 50
					clr_character_flag = multi_animal_skinchanger
				}	
			}
		}
		if = {
			limit = { NOT = { has_character_flag = multi_animal_skinchanger } }
			remove_character_modifier = suspected_skinchanger
			set_variable = { which = "animal_companions" value = 0 }
		}
	}
	
	option = {
		name = "OK"
		
	}
}
