no_tactic = {

		always = no # never use unless set explicitly by code
	}
}

dragon_harass_tactic_light = {
	days = 6
	sprite = 36
	group = dragon

	trigger = {
		phase = skirmish 
		leader = { 
			trait = dragon_rider
			any_friend = {
				trait = dragon
				is_alive = yes
				NOT = { trait = maimed_dragon }
			}
			OR = {
				NOT = { has_character_flag = dragonduel }
				had_character_flag = { flag = dragonduel days = 7 }
				NOT = {
					dragon_rider = {
						is_alive = yes
						at_location = PREV
					}
				}
			}
			NOT = {
				location = { 
					region = world_dorne
					OR = {
						terrain = mountain 
						terrain = desert
					}	
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 27000
		modifier = {
			factor = 0.25
			leader = { 
				trait = dragon_rider
				any_friend = {
					trait = dragon
					is_alive = yes	
					NOT = { trait = maimed_dragon }					
					martial = 20
				}
			}	
		}
	}
	
	knights_offensive = 0.1
	horse_archers_offensive = 0.1
	light_infantry_offensive = 0.1
	heavy_infantry_offensive = 0.1
	pikemen_offensive = 0.1
	archers_offensive = 0.1
	camel_cavalry_offensive = 0.1
	chariots_offensive = 0.1
	war_elephants_offensive = 0.1
	unsullied_offensive = 0.1

	knights_defensive = 0.2
	horse_archers_defensive = 0.2
	light_infantry_defensive = 0.2

	pikemen_defensive = 0.2

	camel_cavalry_defensive = 0.2
	chariots_defensive = 0.2

			NOT = { martial = 8 }
		}
	}

			factor = 1.3

				NOT = { martial = 4 }

			factor = 1.1

				NOT = { martial = 6 }
			}
		}

			factor = 1.1

				NOT = { martial = 8 }

				martial = 5

				}
			}	
		}
		modifier = {
			factor = 1.5

				martial = 7
			}

				}
			}
