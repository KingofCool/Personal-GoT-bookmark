# lifeatcourt.1 - Selling a "Vallerian Sword"
character_event = {
	id = lifeatcourt.1
	desc = "EVTDESClifeatcourt.1" #You hatch up a scheme to sell a fake valerian sword.
	picture = "GFX_smith"
	
	is_triggered_only = yes
	min_age = 14
	max_age = 22
	capable_only = yes
	prisoner = no
	
	trigger = {
		#Generic (Should only trigger for mainland westerosians)
		in_command = no
		OR = {
			culture_group = andal
			culture_group = dornish
			culture_group = first_men
			culture_group = iron_isles_culture
			culture_group = free_cities
			culture = western_valyrian
		}
		NOT = { culture = old_andal }
		NOT = { culture = rhoynar }
		
		#Specific
		NOT = { has_character_flag = tried_to_sell_fake_sword }
		trait = deceitful
		OR = {
			trait = genius
			is_smart_trigger = yes
			trait = ambitious
		}
		NOT = { trait = charitable }
		NOT = { trait = kind }
		NOT = { trait = honest }
		NOT = { trait = honorable }
		NOT = { trait = just }
		NOT = { is_inaccessible_trigger = yes }
		
		#Target/Victim
		location = {
			any_province_character = {
				age = 14
				NOT = { age = 22 }
				OR = {
					trait = trusting
					is_dumb_trigger = yes
					trait = imbecile
					trait = inbred
				}
				NOT_mythical_creature_trigger = yes
				NOT = { is_inaccessible_trigger = yes }
				prisoner = no
				is_incapable = no
				independent = no
				NOT = { character = ROOT }
			}
		}	
	}
	
	weight_multiplier = {
		days = 1
	}
	
	immediate = { set_character_flag = tried_to_sell_fake_sword }
	
	option = {
		name = "Muwahaha!" #Muwahahaha!
		wealth = 50
		location = {
			random_province_character = {
				limit = {
					age = 14
					NOT = { age = 22 }
					OR = {
						trait = trusting
						is_dumb_trigger = yes
						trait = imbecile
						trait = inbred
					}
					NOT_mythical_creature_trigger = yes
					NOT = { is_inaccessible_trigger = yes }
					prisoner = no
					is_incapable = no
					independent = no
					NOT = { character = ROOT }
				}
				opinion = {
					modifier = opinion_furious
					who = ROOT
					years = 10
				}
				hidden_tooltip = { character_event = { id = lifeatcourt.2 } }#Victim gets an event.
			}
		}
	}
}

# lifeatcourt.2 - You are offered to buy a fake valerian sword.
character_event = {
	id = lifeatcourt.2
	desc = "EVTDESClifeatcourt.2" #You are offered to buy a fake valerian sword.
	picture = "GFX_smith"
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAEVTDESClifeatcourt.2" #Pay the blacksmith who evaluated it to keep the whole thing a secret! (Lose 60g, 60% to get shy, lose trusting)
		
		ai_chance = {
			factor = 1
			modifier = {
				factor = 2
				OR = {
					trait = shy
					trait = proud
					trait = lisp
					trait = stutter
					trait = craven
					trait = content
				}
			}
		}
		wealth = -50			
		if = {
			limit = { 
				trait = trusting
			}
			random = {
				chance = 60
				remove_trait = trusting
				hidden_tooltip = { character_event = { id = 38306 } }
			}
		}	
		if = {
			limit = { 
				NOT = { personality_traits = 6 } 
				NOT = { trait = shy }
				NOT = { trait = gregarious }
			}
			random = {
				chance = 25
				add_trait = shy 
				hidden_tooltip = { character_event = { id = 38263 } }
			}
		}
	}

	option = {
		name = "EVTOPTBEVTDESClifeatcourt.2" #Outragious! Complain to the authorities!
		
		ai_chance = {
			factor = 1
			modifier = {
				factor = 2
				OR = {
					trait = brave
					trait = greedy
					trait = wroth
					trait = just
				}
			}
		}
		
		if = {
			limit = {
				liege = { character = FROM }
			}
			FROM = {
				liege = {
					character_event = {
						id = lifeatcourt.3
						days = 1
					}
				}
			}		
		}
		if = {
			limit = {
				NOT = { liege = { character = FROM } }
			}
			liege = {
				character_event = {
					id = lifeatcourt.3
					days = 1
				}
			}
		}
		
		wealth = -50
		prestige = -100
		if = {
			limit = { 
				trait = trusting
			}
			random = {
				chance = 60
				remove_trait = trusting
				hidden_tooltip = { character_event = { id = 38306 } }
			}
		}
	}
	
}

# lifeatcourt.8 - Beating on a weaker kid. (Rob\Training Dummy)

			culture_group = free_cities
			culture = western_valyrian
		}

		NOT = { culture = rhoynar }

			any_province_character = {
				age = 8
				NOT = { age = 17 }
				NOT = { combat_rating = 4.34 }
				NOT = { trait = brave }
				OR = {
					is_weak_trigger = yes
					is_dumb_trigger = yes
					trait = imbecile
					trait = inbred
					trait = dwarf
					trait = craven
				}
				NOT_mythical_creature_trigger = yes
				NOT = { is_inaccessible_trigger = yes }
				NOT = { has_severe_disability_trigger = yes }
				prisoner = no
				is_incapable = no
				independent = no
				NOT = { character = ROOT }
				NOT = { trait = wounded }
				NOT = { is_maimed_trigger = yes }
			}
		}	
	}
	
	weight_multiplier = {
		days = 1
	}
	
	immediate = { 
		set_character_flag = beaten_on_weaker_kid 
		location = {
			random_province_character = {
				limit = {
					age = 8
					NOT = { age = 17 }
					NOT = { combat_rating = 4.34 }

						is_weak_trigger = yes
						is_dumb_trigger = yes
						trait = imbecile
						trait = inbred

						trait = craven
					}

					NOT = { is_inaccessible_trigger = yes }

					independent = no
					NOT = { character = ROOT }
					NOT = { trait = wounded }
					NOT = { is_maimed_trigger = yes }
				}
				save_event_target_as = weaker_kid
			}
		}
	}
	
	option = {
		name = "EVTOPTAEVTDESClifeatcourt.8" #Beat him & rob him of allowance.
		
		ai_chance = {
			factor = 1
			modifier = {
				factor = 2
				OR = {
					trait = slothful
					trait = greedy
				}
			}

				years = 10

			}
		}

			limit = { trait = skilled_warrior }

			}
		}

			}
		}

			}
		}

			limit = {

				NOT = { trait = master_warrior }
			}

			}
		}
