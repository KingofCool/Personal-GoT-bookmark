# The Clergy demands a stop to commerce in all church squares
province_event = {
	id = 77000
	desc = EVTDESC77000
	picture = GFX_evt_market
	
	trigger = {
		NOT = {
			has_province_modifier = stop_commerce
		}
		any_province_lord = {
			is_theocracy = yes
		}
		any_province_lord = {
			is_republic = yes
		}
	}
	
	mean_time_to_happen = {
		months = 36000
	}
	
	option = {
		name = EVTOPTA77000
		owner = {
			piety = 5
		}
		add_province_modifier = {
			name = stop_commerce
			duration = 3650
		}
		random_province_lord = {
			limit = {
				is_theocracy = yes
			}
			opinion = {
				modifier = opinion_happy
				who = ROOT
				years = 5
			}
		}

			limit = {
				is_republic = yes
			}

# The Nobles demands hunting rights on church lands

	trigger = {
		any_province_lord = {
			is_adult = yes

		}
		any_province_lord = {
			NOT = { trait = incapable }
			is_theocracy = yes
			NOT = { ROOT = { owner = { character = PREVPREV } } }
		}
	}
	
	mean_time_to_happen = {
		months = 36000		
	}
	
	immediate = {
		random_province_lord = {

				is_adult = yes

				NOT = { trait = incapable }
				NOT = { ROOT = { owner = { character = PREVPREV } } }
			}
			save_event_target_as = target_feudal
		}

			limit = {
				NOT = { trait = incapable }
				is_theocracy = yes

		name = EVTOPTA77004

##AGOT, important city gains base value
province_event = {
	id = province.1
	desc = EVTDESCprovince.1
	picture = GFX_evt_busy_trading_dock_republic
	
	trigger = {
		#capital of a kingdom
		is_capital = yes
		owner = { 
			higher_tier_than = DUKE
			independent = yes
			OR = {
				tier = EMPEROR
				primary_title = { NOT = { is_variable_equal = { which = "de_facto_empire" value = 1 } } }
			}
			war = no
			NOT = { culture_group = unoccupied_group }
			NOT = { culture_group = winter_group }
		}	
		OR = { #must be a port or on trade route
			port = yes 
			borders_major_river = yes
			has_trade_post = yes		
		}	
		NOR = { #These provinces are already linked to large cities
			province_id = 182 #Casterly Rock
			province_id = 352 #Sunspear
			province_id = 419 #Old Volantis
		}
		any_province_holding = { holding_type = city }
		NOT = {
			any_province_holding = {
				holding_type = city
				base_value_6_trigger = yes
			}
		}
		NOR = {
			has_province_modifier = depopulated_1
			has_province_modifier = depopulated_2
			has_province_modifier = depopulated_3
			has_province_flag = colony_province
			has_province_modifier = overcrowding
		}		
	}
	
	mean_time_to_happen = {
		months = 2000
		
		#Competing city already owned
		modifier = {
			factor = 3
			owner = {
				any_demesne_province = {
					NOT = { province_id = ROOT }

						holding_type = city

			factor = 0.9

			factor = 0.75

		}

			factor = 0.6

			factor = 3

			factor = 1.25
			revolt_risk = 2

			factor = 0.8
