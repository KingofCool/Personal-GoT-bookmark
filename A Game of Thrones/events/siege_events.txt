character_event = {
	id = 62015
	desc = EVTDESC62015
	title = EVTNAME62015
	picture = GFX_evt_crusaders
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			is_ill = yes
			trait = pneumonic
			trait = has_greyscale
			trait = syphilitic
		}
		NOT = {
			trait = craven
		}
	}
	
	weight_multiplier = {
		days = 1
	}
	
	option = {
		name = EVTOPTA62015
		any_defender = {
			troops = -0.2
			morale = -0.2
		}
		troops = -0.1
		random_list = {
			60 = {
				add_maimed_trait_effect = yes
				prestige = 50
			}
			40 = {
				death = {
					death_reason = death_trait
				}
			}
		}
	}
	option = {
		name = EVTOPTB62015
		trigger = { NOT = { trait = berserker } }
		any_defender = {
			troops = -0.05
			morale = -0.05
		}
		troops = -0.15
		if = {
			limit = {
				NOT = {
					trait = craven
				}
				NOT = { trait = brave }
				NOT = { personality_traits = 6 }
			}		
			random = {
				chance = 50

				add_trait = craven
				hidden_tooltip = {
					character_event = {
						id = 38257 #Notify
					}
				}
			}
		}
		prestige = -50
	}
}
