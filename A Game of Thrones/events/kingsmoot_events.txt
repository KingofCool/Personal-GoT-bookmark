character_event = {
	id = kingsmoot.1
	desc = "EVTDESCkingsmoot.1" 
	picture = "GFX_drowned"
	
	is_triggered_only = yes
	
	immediate = {
		#Count candidates
		set_variable = { which = "kingsmoot_candidates" value = 0 }
		
		clr_character_flag = kingsmoot_candidate
		clr_character_flag = candidate_number_found
		clr_character_flag = kingsmoot_candidate_contesting
		clr_character_flag = dynasty
		clr_character_flag = close_relative
		clr_character_flag = kingsmoot_imposter
		#Record laws
		primary_title = {
			#CA
			if = {
				limit = { has_law = centralization_0 }
				set_title_flag = centralization_0
			}
			if = {
				limit = { has_law = centralization_1 }
				set_title_flag = centralization_1
			}
			if = {
				limit = { has_law = centralization_2 }
				set_title_flag = centralization_2
			}
			if = {
				limit = { has_law = centralization_3 }
				set_title_flag = centralization_3
			}
			if = {
				limit = { has_law = centralization_4 }
				set_title_flag = centralization_4
			}
			#Slavery
			if = {
				limit = { has_law = slavery_0 }
				set_title_flag = slavery_0
			}
			if = {
				limit = { has_law = slavery_1 }
				set_title_flag = slavery_1
			}
			if = {
				limit = { has_law = slavery_2 }
				set_title_flag = slavery_2
			}
			if = {
				limit = { has_law = slavery_3 }
				set_title_flag = slavery_3
			}
			if = {
				limit = { has_law = treaty_breach_0 }
				set_title_flag = treaty_breach_0
			}	
			if = {
				limit = { has_law = treaty_breach_1 }
				set_title_flag = treaty_breach_1
			}			
			#First Night
			if = {
				limit = { has_law = first_night_0 }
				set_title_flag = first_night_0
			}
			if = {
				limit = { has_law = first_night_1 }
				set_title_flag = first_night_1
			}
			#Gender Law
			if = {
				limit = { has_law = cognatic_succession }
				set_title_flag = cognatic_succession
			}
			if = {
				limit = { has_law = agnatic_succession }
				set_title_flag = agnatic_succession
			}
			if = {
				limit = { has_law = true_cognatic_succession }
				set_title_flag = true_cognatic_succession
			}
		}	
	}
	
	option = {		
		name = {
			text = "EVTOPTAkingsmoot.1" 
			trigger = {
				prisoner = no
				NOT = { trait = incapable }
				age = 14
			}
		}
		name = {
			text = "EVTOPTAkingsmoot.1B" 
			trigger = {
				OR = {
					prisoner = yes
					trait = incapable
					NOT = { age = 14 }
				}	
			}
		}

			name = holding_large_feast	
			duration = 180
		}

			if = {
				limit = { is_close_relative = FROM }
				set_character_flag = close_relative
			}
			if = {
				limit = { dynasty = FROM }
				set_character_flag = dynasty
			}

				character_event = { id = kingsmoot.2 }
			}
			FROM = {
				any_dynasty_member = {
					limit = {
						is_close_relative = FROM
						is_alive = yes
						is_ruler = no			
						NOT = { any_liege = { character = ROOT } }
					}
					character_event = { id = kingsmoot.2 }
				}
			}

			random = {
				chance = 2

			}
			any_vassal = {
				hidden_tooltip = { character_event = { id = feast.3 } } #random events
			}
		}
	}

character_event = {
	id = kingsmoot.4
	
	is_triggered_only = yes
	
	hide_window = yes
	
	immediate = {	
		#If 6 or less proceed to the election
		if = {
			limit = { NOT = { check_variable = { which = "kingsmoot_candidates" value = 6 } } }
			set_variable = { which = "kingsmoot_candidates" value = 0 }
			set_character_flag = candidates_found
			clr_character_flag = finding_candidates
			character_event = { id = kingsmoot.8 days = 1 } 		
		}	
		#if more than 6 randomly remove candidates
		if = {
			limit = { 
				NOT = { has_character_flag = candidates_found }
				check_variable = { which = "kingsmoot_candidates" value = 6 } 
			} 
			any_realm_character = {
				limit = { has_character_flag = kingsmoot_candidate }	
				character_event = { id = kingsmoot.5 days = 1 random = 2 }
			}
			#set_variable = { which = "kingsmoot_candidates" value = 0 }
			character_event = { id = kingsmoot.4 days = 4 } 
		}
	}
	
	option = {
		name = OK	
	}
}
