# Rebels Kill Random Province Lord - Duke or above
character_event = {
	id = better_rebels.5
	desc = "EVTDESCbetter_rebels.5"
	picture = "GFX_evt_peasants"

	only_playable = yes
	prisoner = no
	
	trigger = {
		higher_tier_than = COUNT
		in_command = no
		# Revolting province
		location = {
			has_province_modifier = brm_revolt
		}
	}
	
	mean_time_to_happen = {
		years = 50
		modifier = {
			factor = 5.0
			NOT = { age = 16 }
		}		
		modifier = {
			factor = 0.5
			NOT = {
				location = {
					any_province_lord = {
						any_demesne_title = {
							location = {
								province_id = PREVPREVPREV
							}
							tier = baron
							NOT = { controlled_by = e_rebels }
						}
					}
				}
			}			
		}		
	}
	
	immediate = {
		hidden_tooltip = {
			random_list = {
				50 = { set_character_flag = character_killed_by_rebels }
				25 = { set_character_flag = character_maimed_by_rebels }
				25 = { set_character_flag = character_wounded_by_rebels }
			}
		}
	}
	
	option = {
		name = "EVTOPTAbetter_rebels.5"
		trigger = { has_character_flag = character_killed_by_rebels }
		hidden_tooltip = {
			if = {
				limit = { independent = no }
				liege = { character_event = { id = better_rebels.9 days = 3 } }
			}
		}
		# Clear flags
		clr_character_flag = character_killed_by_rebels
		clr_character_flag = character_maimed_by_rebels
		clr_character_flag = character_wounded_by_rebels
		#Check if Rhllor saves character
		if = {
			limit = { 
				religion = rhllor 
				OR = {
					trait = resurrected_by_rhllor
					AND = {
						OR = { 	
							trait = favored_by_rhllor
							trait = chosen_by_rhllor				
						}
						OR = {
							check_variable = { which = "red_god_magic" value = 10 }
							has_global_flag = age_of_dragons
						}
					}
				}	
			}
			random = {
				chance = 10
				modifier = {
					factor = 2
					trait = resurrected_by_rhllor
				}
				modifier = {
					factor = 2
					trait = chosen_by_rhllor
				}
				modifier = {
					factor = 2
					has_global_flag = age_of_dragons
				}
				modifier = {
					factor = 1.33
					check_variable = { which = "red_god_magic" value = 1 }
				}
				modifier = {
					factor = 1.5
					check_variable = { which = "red_god_magic" value = 5 }
				}
				character_event = { id = red_priest.38 tooltip = CUSTOMTOOLTIPrhllor_saved_me }
				break = yes
			}
		}
		death = { death_reason = death_rabble }
	}
	option = {
		name = "EVTOPTBbetter_rebels.5"
		trigger = { has_character_flag = character_maimed_by_rebels }
		hidden_tooltip = {
			if = {
				limit = { independent = no }
				liege = { letter_event = { id = better_rebels.10 days = 3 } }
			}
		}
		add_maimed_trait_effect = yes
		# Clear flags
		clr_character_flag = character_killed_by_rebels
		clr_character_flag = character_maimed_by_rebels
		clr_character_flag = character_wounded_by_rebels
	}
	option = {
		name = "EVTOPTCbetter_rebels.5"
		trigger = { has_character_flag = character_wounded_by_rebels }
		hidden_tooltip = {
			if = {
				limit = { independent = no }
				liege = { letter_event = { id = better_rebels.11 days = 3 } }
			}
		}
		add_trait = wounded
		# Clear flags
		clr_character_flag = character_killed_by_rebels
		clr_character_flag = character_maimed_by_rebels
		clr_character_flag = character_wounded_by_rebels
	}
}
