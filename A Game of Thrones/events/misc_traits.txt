character_event = {
	id = misc_traits.2
	desc = "EVTDESCmisc_traits.2"
	picture = "GFX_fashion"
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAmisc_traits.2" #Encourage him.	
		ai_chance = {
			factor = 1
			modifier = {
				factor = 5
				trait = fashionable
			}
			modifier = {
				factor = 2
				trait = lustful
			}
			modifier = {
				factor = 2
				trait = poet
			}
			modifier = {
				factor = 2
				trait = gregarious
			}
			modifier = {
				factor = 2
				trait = kind
			}
			modifier = {
				factor = 2
				trait = patient
			}
		}	
		if = {
			limit = {
				OR = {
					trait = gregarious
					trait = lustful
					trait = poet
					trait = fashionable
					trait = hedonist
				}
			}
			random_list = {
				50 = { change_diplomacy = 1 }

			}
		}

			add_trait = fashionable	
			hidden_tooltip = {		
				character_event = {
					id = misc_traits.3

				}
			}
		}
	}

		name = "EVTOPTBmisc_traits.2" #Tell him to stop it!

				factor = 2
				trait = zealous
			}

				trait = chaste
			}
			modifier = {
				factor = 2
				trait = shy
			}
			modifier = {
				factor = 2
				trait = wroth
			}
			modifier = {
				factor = 2

			}
