# The son of a ruler volunteers to join the Knights Templar (Warrior's Sons)
character_event = {
	id = SoA.1030
	desc = EVTDESC_SOA_1030
	picture = GFX_evt_crusaders
	border = GFX_event_normal_frame_religion
	
	only_men = yes
	min_age = 16
	max_age = 45
	capable_only = yes
	prisoner = no
	religion = the_seven
	has_dlc = "Sons of Abraham"
	war = no
	
	trigger = {
		is_title_active = d_warriors_sons
		#is_ruler = no
		trait = knight
		
		OR = {
			trait = trained_warrior
			trait = skilled_warrior
			trait = master_warrior
		}
		
		NOT = { trait = kingsguard }
		NOT = { trait = nightswatch }
		NOT = { trait = maester }
		
		NOT = { trait = cynical }
		NOT = { is_inaccessible_trigger = yes }
		OR = {
			is_ruler = yes
			father_even_if_dead = {
				higher_tier_than = BARON
			}
			mother_even_if_dead = {
				higher_tier_than = BARON
			}
		}

		liege = {
			holy_order = no
			liege = {
				holy_order = no
			}
		}
				
		NOT = {
			any_spouse = {
				is_pregnant = yes
				father_of_unborn = { character = ROOT } # Won't happen if spouse is pregnant with non-bastard child
			}
		}
		NOT = { has_minor_title = title_triarch }
	}
	
	mean_time_to_happen = {
		years = 275
		modifier = {
			factor = 0.25
			trait = zealous
		}
		modifier = {
			factor = 0.5
			trait = celibate
		}
		modifier = {
			factor = 0.5
			trait = homosexual
		}
		modifier = {
			factor = 0.7
			trait = content
		}
		modifier = {
			factor = 0.75
			trait = chaste
		}
		modifier = {
			factor = 0.75
			NOT = { age = 25 }
		}
		modifier = {
			factor = 1.5
			trait = slothful
		}
		modifier = {
			factor = 4.0
			trait = lustful
		}
		modifier = {
			factor = 4.0
			trait = hedonist
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			is_married = yes
		}
		modifier = {
			factor = 2.0
			any_heir_title = {
				always = yes
			}
		}
		modifier = {
			factor = 3.0
			is_ruler = yes
		}
		modifier = {
			factor = 3.0
			is_primary_heir = yes
		}
		modifier = {
			factor = 3.0
			is_ill = yes
		}
		modifier = {
			factor = 3.0
			trait = stressed
		}
	}
	
	option = {
		name = EVTOPTA_SOA_1030 # Ask to go
		trigger = { is_ruler = no }
		liege = {
			character_event = {
				id = SoA.1031
			}
		}
	}
	
	option = {
		name = EVTOPTB_SOA_1030 # Abdicate
		trigger = { is_ruler = yes }
		piety = 100
		hidden_tooltip = { #Universal on death/abdication event
			if = {
				limit = { has_character_modifier = owns_valyrian_steel_sword }
				set_character_flag = do_not_inherit_sword
			}
			set_character_flag = abdication
			character_event = { id = 45341 } 
			clr_character_flag = abdication
			clr_character_flag = do_not_inherit_sword
			
			any_realm_lord = {
				limit = { ai = no }
				letter_event = { id = SoA.1034 }
			}
			liege = {
				if = {
					limit = { 
						ai = no 
						NOT = { character = ROOT }
					}
					letter_event = { id = SoA.1034 }
					any_vassal = {
						limit = { 
							ai = no 
							NOT = { character = ROOT }
						}
						letter_event = { id = SoA.1034 }
					}
				}	
			}
		}
		if = {
			limit = { is_betrothed = yes }
			betrothed = {
				if = {
					limit = { 
						is_ruler = yes 
						NOT = { character = ROOT }
					}
					opinion = {
						who = ROOT
						modifier = opinion_betrothal_broken
					}
					opinion = {
						who = PREV
						modifier = opinion_betrothal_broken
					}
					if = {
						limit = { 
							ai = no 
							NOT = { same_realm = ROOT }
						}
						letter_event = { id = SoA.1034 }
					}
				}
				if = {
					limit = { 
						is_ruler = no 
						liege = { 
							is_close_relative = PREV 
							NOT = { character = ROOT }
						}
					}
					opinion = {
						who = ROOT
						modifier = opinion_betrothal_broken
					}
					liege = {
						opinion = {
							who = ROOT
							modifier = opinion_betrothal_broken
						}
						opinion = {
							who = PREVPREV
							modifier = opinion_betrothal_broken
						}
						if = {
							limit = { 
								ai = no 
								NOT = { same_realm = ROOT }
							}
							letter_event = { id = SoA.1034 }
						}
					}	
				}
				break_betrothal = PREV
			}	
		}
		if = {
			limit = { is_merchant_republic = no }
			transfer_wealth_to_heir_effect = yes
		}
		abdicate = yes
		any_spouse = {
			remove_spouse = ROOT
		}
		d_warriors_sons = {
			holder_scope = {
				ROOT = {					
					hidden_tooltip = {
						clear_wealth = THIS
					}
					move_character = PREV
					add_trait = celibate
					add_trait = disinherited
				}
			}
		}
		hidden_tooltip = {
			d_poor_fellows = {
				holder_scope = {
					create_random_soldier = {
						random_traits = yes
						dynasty = none
						religion = the_seven
						female = no
						trait = celibate
						trait = knight
					}
					new_character = {
						random_list = {
							40 = { add_trait = poor_warrior }
							50 = { add_trait = trained_warrior }
							10 = { add_trait = skilled_warrior }
						}
						random = {
							chance = 25
							add_trait = zealous
						}
					}			
				}
			}
		}
	}
	
	option = {
		name = EVTOPTC_SOA_1030 # No
		trigger = { ai = no }
	}
}

		ai_chance = {
			factor = 100
			
			modifier = {
				factor = 0.25
				FROM = { is_betrothed = yes }
			}
			modifier = {
				factor = 0
				AND = { 
					current_heir = { character = FROM }
					NOT = { num_of_children = 2 }
				}
			}
			modifier = {
				factor = 0.5
				AND = { 
					current_heir = { character = FROM }
					NOT = { num_of_children = 3 }
				}
			}
			modifier = {
				factor = 0.5
				AND = { 
					current_heir = { character = FROM }
					NOT = { num_of_children = 4 }
				}
			}
		}
		
		if = {
			limit = {
				religion = FROM
			}
			piety = 100
		}
		
		FROM = {
			if = {
				limit = { is_betrothed = yes }
				betrothed = {
					if = {
						limit = { 
							is_ruler = yes 
							NOT = { character = ROOT }

							who = ROOT
							modifier = opinion_betrothal_broken
						}
						opinion = {
							who = PREV
							modifier = opinion_betrothal_broken
						}
					}
					if = {
						limit = { 
							is_ruler = no 
							liege = { 
								is_close_relative = PREV 
								NOT = { character = ROOT }
							}
						}

							who = ROOT
							modifier = opinion_betrothal_broken

			}
