		}
		siege = {
			enemy = {
				leader = {
					save_event_target_as = shadow_baby_target

					AND = {
						OR = {
							trait = envious
							trait = proud
							trait = just
							NOT = { opinion = { who = event_target:shadow_baby_target value = 0 } }
						}
						any_spouse = { is_lover = event_target:shadow_baby_target }				
					}

					trait = lunatic
					trait = possessed

					# Liege...
					AND = {

								}							
							}
							event_target:shadow_baby_target = { #Tyrant
								check_variable = { which = "tyrant" value = 14.9 }
								NOT = { trait = tyrannical_stain }
							}
						}	
					}
										
					# Threats and enemies
					AND = {
						is_ruler = yes
						war_with = event_target:shadow_baby_target			
						OR = {
							is_primary_war_attacker = yes
							is_primary_war_defender = yes
							liege = { character = PREV }
						}
						event_target:shadow_baby_target = {
							is_adult = yes
							NOT = { trait = incapable }
							OR = {
								is_primary_war_attacker = yes
								is_primary_war_defender = yes
							}
						}
					}

							trait = envious

							trait = greedy

								NOT = { is_close_relative = event_target:shadow_baby_target }
							}
							NOT = { opinion = { who = event_target:shadow_baby_target value = 0 } }
						}
						OR = {
							AND = {
								OR = {
									any_pretender_title = { # Get rid of the current ruler or heir if I'm a pretender
										OR = {
											current_heir = { character = event_target:shadow_baby_target }
											holder_scope = { character = event_target:shadow_baby_target }
										}
									}
									any_heir_title = { # Kill a ruler if I'm the heir
										holder_scope = { character = event_target:shadow_baby_target }
										NOT = { has_law = succ_appointment }
									}
								}							
								OR = {
									NOT = { trait = just }
									OR = {
										has_opinion_modifier = { who = event_target:shadow_baby_target modifier = opinion_usurped_title }
										any_pretender_title = {
											current_heir = { character = event_target:shadow_baby_target }
											holder_scope = { reverse_has_opinion_modifier = { who = PREVPREV modifier = opinion_usurped_title } }
										}
									}	
								}
							}
							any_child = { # Gets rid of obstacles in the way of a child's succession
								OR = {
									any_pretender_title = {
										current_heir = { character = event_target:shadow_baby_target }
									}
									any_heir_title = {
										holder_scope = { character = event_target:shadow_baby_target }
									}
								}
								OR = {
									PREV = { NOT = { trait = just } }

										current_heir = { character = event_target:shadow_baby_target }

									}
									has_opinion_modifier = { who = event_target:shadow_baby_target modifier = opinion_usurped_title }
								}
							}
							any_spouse = {	# Gets rid of obstacles in the way of a spouse's succession
								OR = {
									any_pretender_title = { # Get rid of the current heir to a title the character's spouse is a pretender to
										current_heir = { character = event_target:shadow_baby_target }
									}
									any_heir_title = { # Get rid of a ruler the spouse is heir to
										holder_scope = { character = event_target:shadow_baby_target }
									}
								}	
								OR = {
									PREV = { NOT = { trait = just } }
									any_pretender_title = {
										current_heir = { character = event_target:shadow_baby_target }
										holder_scope = { reverse_has_opinion_modifier = { who = PREVPREV modifier = opinion_usurped_title } }
									}
									has_opinion_modifier = { who = event_target:shadow_baby_target modifier = opinion_usurped_title }
								}
							}

					AND = { #These motives are invalid for just characters

							trait = ruthless
							trait = ambitious
							trait = proud
							trait = greedy
							trait = envious
							trait = selfish
							trait = arbitrary
						}

							# Claimant, kill competent holder

								is_ruler = yes

								NOT = { check_variable = { which = "tyrant" value = 9.9 } }

								diplomacy = 8
								intrigue = 8
								NOT = { has_character_modifier = suspected_bastard }								
							}						
							# Powerful and ambitious pretenders
							event_target:shadow_baby_target = {
								OR = {
									any_pretender_title = {
										OR = {
											holder_scope = { character = ROOT }
											current_heir = { character = ROOT }
										}
									}
									any_claim = {
										OR = {
											holder_scope = { character = ROOT }
											current_heir = { character = ROOT }
										}
									}
								}	
								# Must actually be a threat, at least in theory
								is_adult = yes
								NOT = { trait = incapable }
								OR = {
									intrigue = 14
									martial = 14
									trait = dragon_rider
									AND = {
										liege_before_war = { character = ROOT }
										relative_power_to_liege = 0.33
									}

									trait = ambitious
