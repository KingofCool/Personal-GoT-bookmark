#Opinions
character_event = { 
	id = cadet.10
	desc = "EVTDESCcadet.10"
	picture = GFX_evt_courtiers_talking

	only_playable = yes #landed cadet members 
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		OR = {
			has_dynasty_flag = has_cadet_house
			has_dynasty_flag = is_cadet_house
		}
	}
	
	immediate = {
		any_dynasty_member_even_if_dead = {
			limit = { trait = assign_mission_target }
			any_playable_ruler = { #Parent houses
				limit = {
					has_dynasty_flag = has_cadet_house
					has_dynasty_flag = owned_cadet_house_@PREV
					NOT = { dynasty = ROOT }
				}
				if = {
					limit = { NOT = { has_opinion_modifier = { who = ROOT modifier = opinion_cadet_affection } } }
					reverse_opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
					opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
				}
				any_dynasty_member = {
					limit = { NOT = { has_opinion_modifier = { who = ROOT modifier = opinion_cadet_affection } } }
					reverse_opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
					opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
				}
			}
			any_playable_ruler = { #Junior houses
				limit = {
					has_dynasty_flag = is_cadet_house
					has_dynasty_flag = cadet_house_@PREV	
					NOT = { dynasty = ROOT }
				}
				if = {
					limit = { NOT = { has_opinion_modifier = { who = ROOT modifier = opinion_cadet_affection } } }
					reverse_opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
					opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
				}
				any_dynasty_member = {
					limit = { NOT = { has_opinion_modifier = { who = ROOT modifier = opinion_cadet_affection } } }
					reverse_opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
					opinion = { who = ROOT modifier = opinion_cadet_affection years = 200 }
				}
			}
		}
	}

	option = {
		name = "EVTOPTAcadet.10"	
	}
}

character_event = { 
	id = cadet.18

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		trait = bastard	
		NOT = { dynasty = 0 }
		FROM = { is_primary_holder_title = yes }
		OR = {
			father_even_if_dead = { dynasty = ROOT }
			mother_even_if_dead = { dynasty = ROOT }
		}
		NAND = {
			trait = assign_mission_target
			has_dynasty_flag = is_cadet_house
			OR = {
				father_even_if_dead = { has_dynasty_flag = owned_cadet_house_@PREV }
				mother_even_if_dead = { has_dynasty_flag = owned_cadet_house_@PREV }
			}
		}
		NOT = { trait = nightswatch }
		NOT = { trait = kingsguard }
		NOT = { trait = archmaester }
		is_theocracy = no
	}
	
	immediate = {
		#determine if father or mother is cadet founder
		if = {
			limit = { mother_even_if_dead = { dynasty = ROOT } }
			mother_even_if_dead = { save_event_target_as = cadet_founder }
		}
		if = {
			limit = { father_even_if_dead = { dynasty = ROOT } }
			father_even_if_dead = { save_event_target_as = cadet_founder }
		}
		#Mark descendents to change dynasty
		any_child_even_if_dead = {
			limit = { 
				dynasty = ROOT
				NOR = {
					father_even_if_dead = { dynasty = PREV NOT = { character = ROOT } }
					mother_even_if_dead = { dynasty = PREV NOT = { character = ROOT } }
				}
			}
			set_character_flag = change_dynasty
			any_child_even_if_dead = {
				limit = { 
					dynasty = ROOT 
					NOR = {
						father_even_if_dead = { dynasty = PREV NOT = { character = ROOT } NOT = { character = PREV } }
						mother_even_if_dead = { dynasty = PREV NOT = { character = ROOT } NOT = { character = PREV } }
					}
				}
				set_character_flag = change_dynasty
				any_child_even_if_dead = {
					limit = { 
						dynasty = ROOT 
						NOR = {
							father_even_if_dead = { dynasty = PREV NOT = { character = ROOT } NOT = { character = PREV } }
							mother_even_if_dead = { dynasty = PREV NOT = { character = ROOT } NOT = { character = PREV } }
						}
					}
					set_character_flag = change_dynasty
				}
			}
		}
		#create character to get interesting dynasty name
		if = { 
			limit = { dynasty = event_target:cadet_founder }	
			create_character = { 
				random_traits = yes
				dynasty = culture
				religion = THIS
				culture = THIS
				female = no
				age = 30
			}
			new_character = {
				ROOT = { dynasty = PREV }
				dynasty = father_bastard
				death = yes
			}	
		}	
		#Change descendents
		any_child_even_if_dead = {
			limit = { has_character_flag = change_dynasty }
			dynasty = ROOT
			clr_character_flag = change_dynasty
			any_child_even_if_dead = {
				limit = { has_character_flag = change_dynasty }
				dynasty = ROOT
				clr_character_flag = change_dynasty
				any_child_even_if_dead = {
					limit = { has_character_flag = change_dynasty }
					dynasty = ROOT
					clr_character_flag = change_dynasty
				}
			}
		}
		#Set cadet link
		add_trait = assign_mission_target
		set_dynasty_flag = is_cadet_house
		event_target:cadet_founder = {
			add_trait = assign_mission_target
			set_dynasty_flag = has_cadet_house
			set_dynasty_flag = owned_cadet_house_@PREV
			ROOT = { set_dynasty_flag = cadet_house_@PREV }
		}
	}

	option = {
		name = OK	
	}
}

		trigger = { not = { has_global_flag = cadet_45000355 } }
	
		set_global_flag = cadet_45000355
		FROM = { dynasty = 45000355 }
	}
	option = {
		name = "EVTOPTBcadet.57"
		trigger = { not = { has_global_flag = cadet_45000356 } }
	
		set_global_flag = cadet_45000356
		FROM = { dynasty = 45000356 }
	}
	option = {
		name = "EVTOPTCcadet.57"
		trigger = { not = { has_global_flag = cadet_45000357 } }
	
		set_global_flag = cadet_45000357
		FROM = { dynasty = 45000357 }
	}
	after = {
		hidden_tooltip = {
			add_trait = assign_mission_target
			set_dynasty_flag = has_cadet_house		
			FROM = { 
				add_trait = assign_mission_target
				set_dynasty_flag = is_cadet_house
				set_dynasty_flag = cadet_house_@PREV		
				ROOT = { set_dynasty_flag = owned_cadet_house_@PREV }
			}
			reverse_opinion = { who = FROM modifier = opinion_cadet_affection years = 200 }
			opinion = { who = FROM modifier = opinion_cadet_affection years = 200 }
			any_dynasty_member = {
				limit = { NOT = { character = ROOT } }
				reverse_opinion = { who = FROM modifier = opinion_cadet_affection years = 200 }
				opinion = { who = FROM modifier = opinion_cadet_affection years = 200 }
			}
		}	
	}
}
