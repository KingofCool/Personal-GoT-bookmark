			NOT = {
				OR = {
					religion = the_seven
					religion = old_gods

		set_character_flag = studied_religious_myths

			OR = {
				religion = the_seven
				religion = old_gods
				religion_group = wildling_religion
			}
		}
		piety = 20
		hidden_tooltip = { character_event = { id = WoL.7001 days = 90 } }
	}
	option = {
		name = EVTOPTB_WoL_7000 # Yes, listen to stories
		trigger = {
			NOT = {
				OR = {
					religion = the_seven
					religion = old_gods
					religion_group = wildling_religion
				}
			}
		}
		piety = 20
		hidden_tooltip = { character_event = { id = WoL.7001 days = 90 } }

		name = EVTOPTC_WoL_7000  # I have no time
